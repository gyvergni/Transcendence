version: "3.8"
services:
  front:
    build:
      context: ./srcs/front
    volumes:
      - ./dist:/app/dist
    command: sh -c "npm install && npm run build && tail -f /dev/null"
    working_dir: /app
  nginx:
    build:
      context: ./srcs/nginx
    ports:
      - "3000:80"
    depends_on:
      - front
    volumes:
      - ./dist:/usr/share/nginx/html:ro
